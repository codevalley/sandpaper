# Sandpaper v0.3.0 Provider Runtime Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add provider-neutral state and runner contracts, preserve Claude behavior, implement controlled Codex execution, and expose a validated initial provider to the server.

**Architecture:** Small synchronous stores own local preferences and provider/page resume IDs. A static internal registry owns first-party provider identity and diagnostics, while Claude and Codex adapters translate native streams into the existing normalized runner contract. The server remains lifecycle authority and consumes these modules in the linked toolbar plan.

**Tech Stack:** Node.js 18+ ESM, built-in filesystem/process/path APIs, Claude stream-json, Codex JSONL, Node test runner, zero runtime dependencies.

## Global Constraints

- Preserve one process per provider turn and the existing killable runner handle.
- Keep server ownership of turn IDs, concurrency, target hashes, snapshots, Saved/Replied, Undo, terminal cleanup, and reload.
- Store resume IDs by page and provider; never supply one provider's ID to another.
- Migrate legacy `{ "sessionId": "value" }` state to Claude without losing it.
- Reuse saved Codex authentication; remove `CODEX_API_KEY` and `OPENAI_API_KEY` child overrides.
- Codex toolbar execution uses `workspace-write`, approval `never`, no network/web/apps/multi-agent, ignored user config/rules, and the built-in default model.
- Codex runs from the real document directory; do not claim hard single-file isolation.
- Never silently fall back between providers.
- Provider diagnostic `available` means installed, compatible, and authenticated enough to run a turn.
- Preserve zero runtime dependencies and Node.js `>=18`.
- CI uses fake processes and recorded streams only.

```js
// src/session-store.js
import { existsSync, mkdirSync, readFileSync, renameSync, writeFileSync } from 'node:fs';
import { join } from 'node:path';

export const SESSION_VERSION = 2;
const validProvider = (value) => value === 'claude' || value === 'codex';
const validPage = (value) => typeof value === 'string' && value.startsWith('/');

export function createSessionStore(root, { legacyPage = '/' } = {}) {
  const directory = join(root, '.sandpaper');
  const file = join(directory, 'session.json');
  const read = () => {
    if (!existsSync(file)) return { version: SESSION_VERSION, pages: {} };
    try {
      const value = JSON.parse(readFileSync(file, 'utf8'));
      if (value.version === SESSION_VERSION && value.pages && typeof value.pages === 'object') return value;
      if (typeof value.sessionId === 'string' && value.sessionId) {
        return { version: SESSION_VERSION, pages: { [legacyPage]: { claude: { resumeId: value.sessionId } } }, migrated: true };
      }
      return { version: SESSION_VERSION, pages: {} };
    } catch {
      return { version: SESSION_VERSION, pages: {}, corrupt: true };
    }
  };
  const write = (state) => {
    mkdirSync(directory, { recursive: true });
    const temporary = `${file}.tmp-${process.pid}`;
    writeFileSync(temporary, JSON.stringify(state, null, 2) + '\n', { mode: 0o600 });
    renameSync(temporary, file);
  };
  const initial = read();
  if (initial.migrated) {
    delete initial.migrated;
    write(initial);
  }
  const requireKey = ({ page, provider }) => {
    if (!validPage(page) || !validProvider(provider)) throw new TypeError('Invalid session key');
  };
  return {
    get(key) { requireKey(key); return read().pages[key.page]?.[key.provider]?.resumeId || null; },
    set({ page, provider, resumeId }) {
      requireKey({ page, provider });
      if (typeof resumeId !== 'string' || !resumeId) throw new TypeError('Invalid resume ID');
      const state = read();
      if (state.corrupt) throw new Error('Session state is corrupt');
      state.pages[page] ||= {};
      state.pages[page][provider] = { resumeId, updatedAt: new Date().toISOString() };
      write(state);
    },
    clear(key) {
      requireKey(key);
      const state = read();
      if (state.corrupt) throw new Error('Session state is corrupt');
      if (state.pages[key.page]) delete state.pages[key.page][key.provider];
      write(state);
    },
    inspect() { return read(); },
  };
}
```

```js
// src/provider-registry.js
export function createProviderRegistry(entries) {
  const providers = new Map();
  for (const entry of entries) {
    if (!entry || typeof entry.id !== 'string' || typeof entry.runTurn !== 'function') {
      throw new TypeError('Invalid provider entry');
    }
    if (providers.has(entry.id)) throw new Error(`Duplicate provider: ${entry.id}`);
    providers.set(entry.id, Object.freeze(entry));
  }
  return {
    get(id) { return providers.get(id) || null; },
    diagnostics() {
      return [...providers.values()].map((entry) => ({ id: entry.id, label: entry.label, ...entry.diagnose() }));
    },
  };
}
```

```js
export function diagnoseCodex(runCommand = defaultRunCommand) {
  const version = runCommand('codex', ['--version']);
  if (version.status !== 0) return { available: false, compatible: false, authMethod: null, unavailableCode: 'binary_missing' };
  const rootHelp = runCommand('codex', ['--help']);
  const execHelp = runCommand('codex', ['exec', '--help']);
  const resumeHelp = runCommand('codex', ['exec', 'resume', '--help']);
  const login = runCommand('codex', ['login', 'status']);
  const help = `${rootHelp.stdout}\n${execHelp.stdout}\n${resumeHelp.stdout}`;
  const required = ['--ask-for-approval','--sandbox','--json','--ignore-user-config','--ignore-rules','resume'];
  const compatible = required.every((value) => help.includes(value));
  const authMethod = /ChatGPT/i.test(login.stdout) ? 'chatgpt' : /API key/i.test(login.stdout) ? 'api-key' : login.status === 0 ? 'unknown' : null;
  return { available: compatible && authMethod !== null, compatible, authMethod, version: version.stdout.trim(), unavailableCode: compatible && authMethod ? null : compatible ? 'unauthenticated' : 'incompatible' };
}
```
